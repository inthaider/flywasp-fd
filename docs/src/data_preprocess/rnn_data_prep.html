417    def _create_seqs(self, data: np.ndarray, sequence_length: int = 5, index_start: int = 0) -> Tuple[np.ndarray, np.ndarray]:
418        """
419        Creates sequences of length `sequence_length` from the input `data`.
420
421        Parameters
422        ----------
423        data : numpy.ndarray
424            The input data, with shape `(n_samples, n_features)`.
425        sequence_length : int
426            The length of the sequences to create. Defaults to 5.
427
428        Returns
429        -------
430        x : numpy.ndarray
431            The input sequences, with shape `(n_samples - sequence_length, sequence_length, n_features)`.
432        y : numpy.ndarray
433            The target values, with shape `(n_samples - sequence_length,)`.
434        target_indices : numpy.ndarray
435            The indices of the target values in the original data.
436
437        Notes
438        -----
439        This function creates sequences of length `sequence_length` from the input `data`. Each sequence consists of `sequence_length`
440        consecutive rows of `data`, and the target value for each sequence is the value in the last row of the sequence.
441        """
442        n = len(data) - sequence_length
443        # -1 because we're dropping the target column in x
444        x = np.empty((n, sequence_length, data.shape[1]-1))
445        y = np.empty(n)
446        target_indices = np.empty(n, dtype=int)
447
448        valid_idx = 0
449        for i in range(n):
450            target = data[i + sequence_length, -1]
451            # -1 because we're dropping the target column in the X sequences
452            sequence = data[i:i + sequence_length, :-1]
453
454            # Check if there are any missing values in the sequence or target
455            if not np.isnan(sequence).any() and not np.isnan(target):
456                x[valid_idx] = sequence
457                y[valid_idx] = target
458
459                # Store the index of the target row/observation in the sequence
460                target_indices[valid_idx] = index_start + i + sequence_length
461
462                valid_idx += 1
463
464        # Trim the arrays to the size of valid sequences
465        x = x[:valid_idx]
466        y = y[:valid_idx]
467        target_indices = target_indices[:valid_idx]
468        return x, y, target_indices

471        """

473
