121        -------
122        str
123            The path to the saved pickled file.
124        """
125        # Logging statement to indicate the start of saving processed data
126        logger.info(
127            "\nSaving processed data in DataPreprocessor -> save_processed_data() ...")
128        try:
129            # Create the directory for the processed data if it doesn't exist
130            processed_data_dir = Path(f"data/processed/{self.raw_data_id}")

133            #

139            # Generating the hash using the md5 algorithm based on the dataframe summary
140            processed_data_hash = hashlib.md5(df_summary.encode()).hexdigest()
141            logger.info("Processed data hashed.")
142
143            # Construct the output file path
144            self.processed_data_path = processed_data_dir / \
145                f"{self.timestamp}_processed_data_{processed_data_hash}.pkl"
146
147            # Save the processed data to the output file
148            logger.info(
149                f"Saving processed data to {self.processed_data_path}...")
